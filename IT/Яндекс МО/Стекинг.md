```python
from sklearn.model_selection import train_test_split
from sklearn.tree import DecisionTreeClassifier # –∫–ª–∞—Å—Å –º–æ–¥–µ–ª–∏ –¥–µ—Ä–µ–≤–∞
from sklearn.ensemble import AdaBoostClassifier, RandomForestClassifier, StackingClassifier # –∫–ª–∞—Å—Å—ã –∞–Ω—Å–∞–º–±–ª–µ–π
from sklearn.metrics import accuracy_score

# –∑–∞–≥—Ä—É–∂–∞–µ–º –∏ –≥–æ—Ç–æ–≤–∏–º –¥–∞–Ω–Ω—ã–µ
df = pd.read_csv('https://yastatic.net/s3/lyceum/files/cef58ceb-187c-4104-9a90-124714292af0/upload.csv', 
                 names=['buying', 'maint', 'doors', 'persons', 'lug_boot', 'safety', 'accep'])
df['accep'] = ~(df['accep'] == 'unacc')  # 1 - –ø—Ä–∏–µ–º–ª–µ–º–æ, 0 - –Ω–µ–ø—Ä–∏–µ–º–ª–µ–º–æ
X = pd.get_dummies(df.iloc[:, 0:6], drop_first=True)

# —Ä–∞–∑–±–∏–≤–∞–µ–º –Ω–∞ —Ç—Ä–µ–Ω–∏—Ä–æ–≤–æ—á–Ω—É—é –∏ —Ç–µ—Å—Ç–æ–≤—É—é –≤—ã–±–æ—Ä–∫–∏
```

```python
# —Å–æ–∑–¥–∞–µ–º –±–∞–∑–æ–≤—ã–µ –º–æ–¥–µ–ª–∏: —Å–ª—É—á–∞–π–Ω—ã–π –ª–µ—Å –∏ –∞–¥–∞–ø—Ç–∏–≤–Ω—ã–π –±—É—Å—Ç–∏–Ω–≥
base_estimators = [
    ('rf', RandomForestClassifier(n_estimators=10, random_state=0)),
    ('gb', AdaBoostClassifier(n_estimators=10, random_state=0))
]
```

```python
stacking_clf = StackingClassifier(
    estimators=base_estimators,
    final_estimator=meta_model,
    passthrough=True  # –¥–æ–±–∞–≤–ª—è–µ—Ç –∏—Å—Ö–æ–¥–Ω—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏ –∫ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è–º –±–∞–∑–æ–≤—ã—Ö –º–æ–¥–µ–ª–µ–π
)

# –ø—Ä–µ–¥—Å–∫–∞–∑—ã–≤–∞–µ–º –Ω–∞ —Ç–µ—Å—Ç–æ–≤—ã—Ö –¥–∞–Ω–Ω—ã—Ö
```
